## Problem to Fix

The review cards currently show a full blue border around the entire card. This is incorrect.

The correct design is:
- **Left side only:** 4px solid colored border (color based on score)
- **Other three sides:** 1px extremely light gray (#f1f5f9) — barely visible
- **No full outline, no ring, no shadow border**

---

## Score Color Rules

Calculate the overall score as the average of the three dimensions:
```
overall = (rating_value + rating_effectiveness + rating_punctuality) / 3
```

Then apply border color:
- overall ≥ 4.0 → `#1668c8` (blue)
- overall 2.5–3.9 → `#c89000` (amber)
- overall < 2.5 → `#c83030` (red)

---

## Step 1: Update ReviewCard Component

Replace `components/review/ReviewCard.tsx` with the following exact implementation:

```typescript
import { Review } from '@/types/database'

interface ReviewCardProps {
  review: Review
}

function scoreColor(score: number): string {
  if (score >= 4.0) return '#1668c8'
  if (score >= 2.5) return '#c89000'
  return '#c83030'
}

function scoreBg(score: number): string {
  if (score >= 4.0) return '#e8f1fc'
  if (score >= 2.5) return '#fdf3dc'
  return '#fde8e8'
}

export function ReviewCard({ review }: ReviewCardProps) {
  const overall =
    (review.rating_value + review.rating_effectiveness + review.rating_punctuality) / 3
  const color = scoreColor(overall)
  const bg = scoreBg(overall)

  return (
    <div
      className="bg-white rounded-xl p-4"
      style={{
        borderLeft: `4px solid ${color}`,
        borderTop: '1px solid #f1f5f9',
        borderRight: '1px solid #f1f5f9',
        borderBottom: '1px solid #f1f5f9',
      }}
    >
      {/* Top row: overall score pill + verified badge */}
      <div className="flex items-center justify-between mb-3">
        <span
          className="inline-flex items-center gap-1.5 text-sm font-bold px-3 py-1 rounded-full"
          style={{ background: bg, color: color }}
        >
          ⭐ {overall.toFixed(1)}
        </span>
        {review.is_verified ? (
          <span className="text-xs font-semibold px-3 py-1 rounded-full bg-green-50 text-green-700">
            ✓ Verified
          </span>
        ) : (
          <span className="text-xs font-semibold px-3 py-1 rounded-full bg-slate-100 text-slate-400">
            Unverified
          </span>
        )}
      </div>

      {/* Three dimension scores */}
      <div className="flex gap-4 flex-wrap mb-3 text-xs font-semibold text-slate-500 uppercase tracking-wide">
        <span>
          💰 Value:{' '}
          <strong className="text-slate-700">{review.rating_value.toFixed(1)}</strong>
        </span>
        <span>
          📈 Effectiveness:{' '}
          <strong className="text-slate-700">{review.rating_effectiveness.toFixed(1)}</strong>
        </span>
        <span>
          ⏰ Punctuality:{' '}
          <strong className="text-slate-700">{review.rating_punctuality.toFixed(1)}</strong>
        </span>
      </div>

      {/* Optional text comment — only shown if present */}
      {review.comment && review.comment.trim() !== '' && (
        <p className="text-sm text-slate-600 leading-relaxed pl-3 border-l-2 border-slate-200">
          {review.comment}
        </p>
      )}
    </div>
  )
}
```
