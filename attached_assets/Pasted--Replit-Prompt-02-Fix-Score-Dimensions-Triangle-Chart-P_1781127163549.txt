## Step 2: Create Triangle Radar Chart Component

Create `components/shared/ScoreTriangle.tsx` with the following exact implementation:

```typescript
'use client'

import { useEffect, useRef } from 'react'

interface ScoreTriangleProps {
  value: number        // 💰 Value score (0–5)
  effectiveness: number // 📈 Effectiveness score (0–5)
  punctuality: number  // ⏰ Punctuality score (0–5)
  reviewCount?: number
}

export function ScoreTriangle({ 
  value, 
  effectiveness, 
  punctuality, 
  reviewCount 
}: ScoreTriangleProps) {
  const CX = 110, CY = 105, R = 80
  const ANGLES = [-90, 30, 150] // top, bottom-right, bottom-left
  const LABELS = ['💰 Value', '📈 Effectiveness', '⏰ Punctuality']
  const LABEL_OFFSETS = [
    { dx: 0, dy: -18 },
    { dx: 22, dy: 14 },
    { dx: -22, dy: 14 },
  ]
  const scores = [value, effectiveness, punctuality]
  const overall = ((value + effectiveness + punctuality) / 3).toFixed(1)

  function getPoint(angleDeg: number, radius: number) {
    const rad = angleDeg * Math.PI / 180
    return { x: CX + radius * Math.cos(rad), y: CY + radius * Math.sin(rad) }
  }

  function polyStr(points: { x: number; y: number }[]) {
    return points.map(p => `${p.x},${p.y}`).join(' ')
  }

  // Grid triangles (5 levels)
  const gridLevels = [1, 2, 3, 4, 5].map(g =>
    polyStr(ANGLES.map(a => getPoint(a, R * g / 5)))
  )

  // Data polygon
  const dataPts = ANGLES.map((a, i) => getPoint(a, R * scores[i] / 5))

  // Score color
  function scoreColor(s: number) {
    if (s >= 4.0) return '#1668c8'
    if (s >= 2.5) return '#c89000'
    return '#c83030'
  }

  const overallColor = scoreColor(parseFloat(overall))

  return (
    <div className="bg-white rounded-2xl border border-slate-100 p-6">
      <p className="text-xs font-semibold text-slate-400 uppercase tracking-widest text-center mb-4">
        Score Breakdown
      </p>

      {/* Triangle SVG */}
      <div className="flex justify-center mb-4">
        <svg width="220" height="210" viewBox="0 0 220 210">
          <defs>
            <linearGradient id="fillGrad" x1="0%" y1="0%" x2="100%" y2="100%">
              <stop offset="0%" stopColor="#1668c8" stopOpacity="0.25"/>
              <stop offset="100%" stopColor="#b8d400" stopOpacity="0.20"/>
            </linearGradient>
          </defs>

          {/* Grid */}
          {gridLevels.map((pts, i) => (
            <polygon key={i} points={pts} fill="none" stroke="#e8f1fc" strokeWidth="1"/>
          ))}

          {/* Axes */}
          {ANGLES.map((a, i) => {
            const tip = getPoint(a, R)
            return <line key={i} x1={CX} y1={CY} x2={tip.x} y2={tip.y} stroke="#dce3ed" strokeWidth="1"/>
          })}

          {/* Data fill */}
          <polygon
            points={polyStr(dataPts)}
            fill="url(#fillGrad)"
            stroke="#1668c8"
            strokeWidth="2"
            strokeLinejoin="round"
          />

          {/* Dots */}
          {dataPts.map((pt, i) => (
            <circle key={i} cx={pt.x} cy={pt.y} r="5" fill="#1668c8" stroke="white" strokeWidth="2"/>
          ))}

          {/* Score labels near dots */}
          {dataPts.map((pt, i) => (
            <text
              key={i}
              x={pt.x + LABEL_OFFSETS[i].dx * 0.6}
              y={pt.y + LABEL_OFFSETS[i].dy * 0.6 - 8}
              textAnchor="middle"
              fontSize="11"
              fontWeight="700"
              fill="#1668c8"
              fontFamily="-apple-system,sans-serif"
            >
              {scores[i].toFixed(1)}
            </text>
          ))}

          {/* Axis labels */}
          {ANGLES.map((a, i) => {
            const tip = getPoint(a, R + 20)
            return (
              <text
                key={i}
                x={tip.x + LABEL_OFFSETS[i].dx}
                y={tip.y + LABEL_OFFSETS[i].dy}
                textAnchor="middle"
                fontSize="12"
                fontWeight="600"
                fill="#1e2a38"
                fontFamily="-apple-system,sans-serif"
              >
                {LABELS[i]}
              </text>
            )
          })}
        </svg>
      </div>

      {/* Overall score */}
      <div className="text-center mb-5 pb-5 border-b border-slate-100">
        <div className="text-5xl font-bold" style={{ color: overallColor }}>
          {overall}
        </div>
        <div className="text-xs text-slate-400 mt-1">
          Overall Score{reviewCount ? ` · ${reviewCount} reviews` : ''}
        </div>
      </div>

      {/* Score bars */}
      <div className="flex flex-col gap-3">
        {[
          { label: '💰 Value', score: value },
          { label: '📈 Effectiveness', score: effectiveness },
          { label: '⏰ Punctuality', score: punctuality },
        ].map(({ label, score }) => (
          <div key={label} className="flex items-center gap-3">
            <span className="text-xs font-semibold text-slate-500 w-28 flex-shrink-0">
              {label}
            </span>
            <div className="flex-1 h-2 bg-blue-50 rounded-full overflow-hidden">
              <div
                className="h-full rounded-full bg-rita-blue transition-all duration-500"
                style={{ width: `${(score / 5) * 100}%` }}
              />
            </div>
            <span className="text-sm font-bold text-rita-blue w-8 text-right">
              {score.toFixed(1)}
            </span>
          </div>
        ))}
      </div>
    </div>
  )
}
```

## Step 3: Update ScoreDisplay Component

Replace `components/shared/ScoreDisplay.tsx` with a simple score pill component:

```typescript
interface ScoreDisplayProps {
  score: number
  size?: 'sm' | 'md' | 'lg'
}

export function ScoreDisplay({ score, size = 'md' }: ScoreDisplayProps) {
  function scoreColor(s: number) {
    if (s >= 4.0) return '#1668c8'
    if (s >= 2.5) return '#c89000'
    return '#c83030'
  }

  const sizes = {
    sm: 'text-sm px-2 py-0.5',
    md: 'text-base px-3 py-1',
    lg: 'text-2xl px-4 py-2',
  }

  return (
    <span
      className={`font-bold rounded-full inline-block ${sizes[size]}`}
      style={{
        color: scoreColor(score),
        background: scoreColor(score) + '18',
      }}
    >
      ⭐ {score.toFixed(1)}
    </span>
  )
}
```
