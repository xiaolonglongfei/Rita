interface InstructorBadgeProps {

}

export function InstructorBadge({ isClaimed }: InstructorBadgeProps) {
  if (isClaimed) {
    return (
      <span className="inline-flex items-center gap-1.5 px-3 py-1 rounded-full text-xs font-semibold bg-rita-blue-light text-rita-blue border border-blue-200">
        ✓ Claimed
      </span>
    )
  }
  return (
    <span className="inline-flex items-center gap-1.5 px-3 py-1 rounded-full text-xs font-semibold bg-slate-100 text-slate-500 border border-slate-200">
      Unclaimed
    </span>
  )
}
